initevents
set %cel #ltargetid
set %profile 350_300

start:

finditem PBG C;
set #lobjectid #findid
finditem MMF
if #findkind = -1
{
set #sysmsgcol 1080
event sysmessage NIE MAM JUZ ZWOJKOW
sound C:\oneandonly.wav
halt
}
else
set #ltargetid #findid
event macro 17 0
target 100s
event macro 22 0 
wait 10

inskr:
if #CONTSIZE = %profile
{
click 105 278
goto pat
}
else
{
goto inskr
}

pat:
if Stworzyles_zwoj in #sysmsg
{
wait 10
goto mana
}
if Nauczasz in #sysmsg
{
wait 10
goto mana
}
if Nie_udalo in #sysmsg
{
wait 10
goto mana
}
if Masz_za in #sysmsg
{
wait 10
goto mana
}
if Your_skill in #sysmsg
{
wait 10
goto mana
}

else
goto pat

mana:
wait 20
if #mana < 40 ;TU WPISUJESZ ILE MANY ZABIERA CZAR
{
event macro 13 46
goto sprawdzmedit
}
else
{
goto dalej
}

sprawdzmedit:
if Konczysz_medytacje in #sysmsg
{
wait 5
goto dalej
}
if Nie_mozesz in #sysmsg
{
wait 6s
goto mana
}
if You_must in #sysmsg
{
wait 3s
goto mana
}
else
{
goto sprawdzmedit
}

dalej:
wait 20
event macro 13 21
goto bless



bless:
if #str = 48   ;tu wstawiasz wartosc swojej sily zeby pisal na blessie
{
event macro 15 16
target 100s
event macro 23
wait 4s
goto mana
}
else
{
goto start
}
