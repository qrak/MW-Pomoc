{
   set %tak 0
   set %tak1 0
   set %nie 0
   set %nie1 0
}

   set #sysmsgcol 1082

   set %nie 0
   set %nie1 0

   set %tak1 %tak1 + 1

   set %tak1 %tak1 + 1
