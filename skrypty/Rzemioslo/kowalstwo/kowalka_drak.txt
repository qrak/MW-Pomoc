   }

stolarz:

{
   if %zmiana = 1
   {
      set %czas #scnt2 + 20
      click %klik1 %klik2
      wait 5
      screen2:
      if #CONTSIZE = %profile
      {
         click %klik3 %klik4
         set %zmiana = 0
         target 1s
         event macro 22 0
         set %czas #scnt2 + 70
         goto pat
      }
      if %czas < #scnt2
         goto pat
      else
          goto pat
   }

   set %czas #scnt2 + 70
   goto pat

pat:

if Zniszczyles_troche in #journal || Zniszczylas_troche in #journal || Wykulas_przemdiot in #journal || Wykules_przedmiot in #journal
{
   goto hid
}

{
   set %gdzie %gdzie + 1
   set %zmiana 1
   goto tnij
}
if Masz_za in #journal
{
   set %skora 1
   goto tnij
}
if W_plecaku_nie_ma in #journal
   goto tnij
if %czas < #scnt2
   goto hid
else
   goto pat
hid:

   {
   event macro 1 0 .um ukrywanie

   goto hid2
   }

   event macro 1 0 .um wykrywanie
   set %czas #scnt2 + 10
   goto hid3
   }
