initevents
set %Hiding 0
ignoreitem reset
ignoreitem #charid
start:
if #hits < #maxhits
{
finditem ZLF
if #findKind = -1
  goto anatomia
if #findKind = 0
{
  healing:
  if #targCurs = 1
   goto healing
  if %LockTime > #scnt
   goto anatomia
  set #lObjectID #FindID
  set #lTargetID #CharID
  set #lTargetKind 1
  event macro 17
  target
  event macro 22
  set %a 0
  sprawdz:
  set %a %a + 1
  wait 5
  scanJournal 1
  if Wyleczyles in #journal || Nie_udalo_Ci in #journal || Przestales in #journal || Stoisz_za in #journal || Nie_mozesz in #journal || %a = 75
  {
   if Nie_mozesz in #journal
    set %LockTime #scnt + 5
   goto anatomia
  }
  else
   goto sprawdz
}
}
else
{
anatomia:
if #targCurs = 1
  goto anatomia
set %czas #scnt - %LastTime
if %czas > 10
{
  finditem * G_3
  str Len #findtype
  if #strRes = 2
  {
   set %LastTime #scnt
   set #lTargetID #findID
   set #lTargetKind 1
   event macro 13 1
   target 4s
   event macro 22
  }
  else
  {
   ignoreitem #findid
   if %Hiding = 1
   {
    set %hid %hid + 1
    if %hid = 10
    {
     set %LastTime #scnt
     set %hid 0
     event macro 13 21
    }
    else
     wait 1
   }
  }
  goto start
}
else
  goto start
}
