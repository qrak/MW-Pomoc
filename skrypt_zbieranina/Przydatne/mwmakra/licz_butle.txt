start:

reg1:
  menu Font color Yellow
  finditem UUF C_ , #backpackid ;
  if %reg1 = #FINDSTACK
  goto reg2
  set %reg1 #FINDSTACK

  menu Font color Red
  menu delete reg1
  menu Text reg1 50 0 %reg1
  
reg2:
  menu Font color Yellow
  finditem NUF C_ , #backpackid ;
  if %reg2 = #FINDSTACK

  set %reg2 #FINDSTACK
  if %reg2 < 6

reg3:
  menu Font color Yellow
  finditem YUF  C_ , #backpackid ;

  set %reg3 #FINDSTACK

reg4:
  menu Font color Yellow
  finditem XUF C_ , #backpackid ;

  goto reg5
  set %reg4 #FINDSTACK
  if %reg4 < 6

  menu Text reg4 50 60 %reg4

reg6:

  finditem ZLF C_ , #backpackid ;

  menu Font color Red
  menu delete reg7
  menu Text reg7 125 40 %reg7
  

goto start
