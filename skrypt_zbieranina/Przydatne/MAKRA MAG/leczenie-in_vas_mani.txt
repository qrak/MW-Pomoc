;Jezeli mamy ponizej '%hp1' hp leczy nas i medytuje
;Zarzuca blesa, prota i reactiva
;Po przytrzymaniu klawisza "insert" skrypt sie wylacza

;U W A G A !!! USTAW PONIZSZE ZMIENNE PRZED WLACZENIEM SKRYPTU !!!
set %bles 1                   ;Tu ustaw czy skrypt ma zarzucac blesa
set %prot 1                   ;Tu ustaw czy skrypt ma zarzucac prota
set %reactiv 1                ;Tu ustaw czy skrypt ma zarzucac reactiv armor

set %sciezka c:\ringin.wav       ;Tutaj ustaw sciezke do dzwieku
;------------------------------------------------------------------------------------------------------------

initevents
set #sysmsgcol 1281
event sysmessage Skrypt [leczenie-in_vas_mani] odpalony (by SabiK)

start:
if #charghost = YES
{
   set #sysmsgcol 1080

   halt
}

{
   set #sysmsgcol 1080
   event sysmessage Wylaczono skrypt [leczenie-in_vas_mani]
   halt
}

if #hits < %hpk

if #hits < %hp
   goto lecz

      event sysmessage Jestes marty :P. Wylaczono skrypt [leczenie-in_vas_mani]
      halt
   }

   set %y #charposy
   set %z #charposz
   set %ch #chardir
   wait 10
   onhotkey insert
   {
      set #sysmsgcol 1080
      event sysmessage Wylaczono skrypt [leczenie-in_vas_mani]
      halt
   }
   if #charposx = %x
   {
      if #charposy = %y
      {
         if #charposz = %z
         {
            if #chardir = %ch
               goto leczenie1
         }
      }
   }
   goto lecz
   leczenie1:
   event macro 15 28
   set %licz #scnt2 + 60
   czekaj_leczenie:
   if #targcurs = 1
      goto dalej_leczenie
   if #sex = 0
   {
      if Straciles_koncentracje in #sysmsg
         goto lecz
   }
   if #sex = 1
   {
      if Stracilas_koncentracje in #sysmsg
         goto lecz
   }
   if #sysmsg = You_are_already_doing_something_else.
   {
      wait 10
      goto lecz
   }
   if #charghost = YES
   {
      set #sysmsgcol 1080
      event sysmessage Jestes marty :P. Wylaczono skrypt [leczenie-in_vas_mani]
      halt
   }
   onhotkey insert
   {
      set #sysmsgcol 1080
      event sysmessage Wylaczono skrypt [leczenie-in_vas_mani]
      halt
   }
   if brakuje in #sysmsg
   {
      set #sysmsgcol 1080
      event sysmessage Brakuje Ci skladnikow lub many! Skrypt wylaczono!
      sound %sciezka
      wait 35
      sound %sciezka
      halt
   }
   if %licz < #scnt2
      goto lecz
   goto czekaj_leczenie
   dalej_leczenie:
   if #hits < %hp1
      goto dalej2_leczenie
   if #targcurs = 0
   {
      wait 5
      if #mana > ( #maxmana - 10 )
      goto start
      else
      goto medyt
   }
   event macro 1 0 .um wykrywanie
   wait 15
   if #hits < %hp1
      goto dalej2_leczenie
   if #targcurs = 0
   {
      wait 5
      if #mana > ( #maxmana - 10 )
      goto start
      else
      goto medyt
   }
   wait 15
   if #hits < %hp1
      goto dalej2_leczenie
   if #targcurs = 0
   {
      wait 5
      if #mana > ( #maxmana - 10 )
      goto start
      else
      goto medyt
   }
   if #charghost = YES
   {
      set #sysmsgcol 1080
      event sysmessage Jestes marty :P. Wylaczono skrypt [leczenie-in_vas_mani]
      halt
   }
   onhotkey insert
   {
      set #sysmsgcol 1080
      event sysmessage Wylaczono skrypt [leczenie-in_vas_mani]
      halt
   }
   goto dalej_leczenie
}
goto start
